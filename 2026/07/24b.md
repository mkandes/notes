# hpcGPT-cli deployment on Expanse

```
npm install -g npm
```

```
npm install -g opencode-ai
```

```
npm config set allow-scripts=opencode-ai --location=user
```

```
npm install -g opencode-plugin-litellm
```

```
npm config set allow-scripts=msgpackr-extract --location=user
```

```
export OPENCODE_EXPERIMENTAL='false'
```

```
export OPENCODE_ENABLE_EXA='true'
```

```
export TRITONAI_API_KEY=XXXXXXXXXX
```

```
mkdir -p "${HOME}/.config/opencode/agents"
```

```
cp ~/software/hpcgpt-cli/SDSC/opencode.jsonc ~/.config/opencode/
```

```
opencode
```

```
rm -rf ~/.bun ~/.cache/opencode ~/.config/opencode ~/.local/share/opencode .local/state/opencode/
```
