#!/usr/bin/env python3
#
# Convert Doccano .jsonl to Label Studio .json and vice versa

import argparse
import json
import logging
import os
import pathlib
import sys
import time

logging.basicConfig(
    level=logging.INFO,
    format="[%(asctime)s.%(msecs)03d] %(levelname)s: %(message)s",
    datefmt='%Y-%m-%d %H:%M:%S'
)


def get_command_arguments():
    """ Read input variables and parse command-line arguments """

    parser = argparse.ArgumentParser(
        description='Convert Doccano .jsonl to Label Studio .json and vice versa',
        formatter_class=argparse.ArgumentDefaultsHelpFormatter,
    )

    parser.add_argument(
        '-f', '--file', type=pathlib.Path, default=None,
        help='path to a .jsonl or .json file'
    )

    args = parser.parse_args()
    return args

def main():
    """ Convert Doccano .jsonl to Label Studio .json and vice versa """

    args = get_command_arguments()

    file = args.file

    try:
        with open(file, 'r', encoding='utf-8') as input_file:
            input_data = json.load(input_file)
    except json.JSONDecodeError as e:
        logging.info(f'{input_file.name} is not a JSON file')
        try:
            with open(file, 'r', encoding='utf-8') as input_file:
                input_lines = []
                for line in input_file:
                    clean_line = line.strip()
                    if clean_line:
                        input_data = json.loads(clean_line)
                        input_lines.append(input_data)
        except json.JSONDecodeError as e:
            logging.info(f'{input_file.name} is not a JSONL file')
            logging.exception(e)
            return 1
        else:
            output_lines = []
            for line in input_lines:
                output_data = {
                    "id": line.pop("id"),
                    "data": {
                        "text": line.pop("text")
                    }
                }
                output_lines.append(output_data)
            try:
                with open(str(file.resolve().parent) + '/' 
                          + str(file.stem) + '.json', 'w', 
                          encoding='utf-8') as output_file:
                    json.dump(output_lines, output_file, indent=4)
            except Exception as e:
                logging.exception(e)
                return 1
    except FileNotFoundError as e:
        logging.exception(e)
        return 1
    else:
        print('convert to jsonl')

    return 0


if __name__ == '__main__':
    sys.exit(main())

# https://labelstud.io
# https://labelstud.io/blog/understanding-the-label-studio-json-format
