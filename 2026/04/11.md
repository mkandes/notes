# Summerization with TritonAI API

```
from openai import OpenAI

client = OpenAI() # Assumes API key is in environment variable OPENAI_API_KEY

response = client.chat.completions.create(
    model="gpt-4o",
    messages=[
        {"role": "system", "content": "Explain concepts simply, like I am five."}, # Your instruction (prompt)
        {"role": "user", "content": "What is the OpenAI API?"} # Your specific question
    ]
)

# Extract and print the answer
print(response.choices[0].message.content)

```
