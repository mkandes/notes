def pre_process_imagenet_pytorch(img, dims=None, need_transpose=False):
    from PIL import Image
    import torchvision.transforms.functional as F

    img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
    img = Image.fromarray(img)
    img = F.resize(img, 256, Image.BILINEAR)
    img = F.center_crop(img, 224)
    img = F.to_tensor(img)
    img = F.normalize(
        img, mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225], inplace=False
    )
    if not need_transpose:
        img = img.permute(1, 2, 0)  # NHWC
    img = np.asarray(img, dtype="float32")
    return img

```
import cv2
import numpy as np

# 1. Convert BGR to RGB
img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)

# 2. Resize maintaining aspect ratio such that the shorter side is 256
h, w = img.shape[:2]
if h < w:
    new_h, new_w = 256, int(w * (256 / h))
else:
    new_h, new_w = int(h * (256 / w)), 256

img_resized = cv2.resize(img, (new_w, new_h), interpolation=cv2.INTER_LINEAR)

# 3. Center crop to 224x224
start_x = (new_w - 224) // 2
start_y = (new_h - 224) // 2
img_cropped = img_resized[start_y : start_y + 224, start_x : start_x + 224]

# 4. Convert to float32 and scale pixel values to [0, 1]
img_float = img_cropped.astype("float32") / 255.0

# 5. Normalize (Mean and Std)
mean = np.array([0.485, 0.456, 0.406], dtype="float32")
std = np.array([0.229, 0.224, 0.225], dtype="float32")
img_normalized = (img_float - mean) / std

# 6. Transpose to NHWC if not need_transpose
if not need_transpose:
    img = img_normalized
else:
    # If the original PyTorch pipeline kept it as NCHW, transpose it to (C, H, W)
    img = np.transpose(img_normalized, (2, 0, 1))

```

import cv2

def center_crop(img, crop_width, crop_height):
    # Get original image dimensions
    height, width = img.shape[:2]
    
    # Safely handle requested dimensions that exceed the original size
    crop_width = min(crop_width, width)
    crop_height = min(crop_height, height)

    # Calculate top-left starting points
    start_x = (width - crop_width) // 2
    start_y = (height - crop_height) // 2

    # Return sliced image array
    return img[start_y:start_y + crop_height, start_x:start_x + crop_width]

# Example usage
image = cv2.imread("your_image.jpg")
# Crop a 400x400 square from the center
cropped_image = center_crop(image, 400, 400)
