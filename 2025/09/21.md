# Function to build a command array with environment variables after parsing them from the input
build_cmd_with_env_vars() {
  # Store the input command as an array
  local input_cmd=("$@")
  local -a env_vars
  local -a prog_and_args
  local key_value
  local processing_envs=0
  local osp_placeholder_index=-1

  # Iterate over the command array to process each item
  for i in "${!input_cmd[@]}"; do
    if [[ "${input_cmd[$i]}" == "--env" ]]; then
      processing_envs=1
      # Split the environment variables string into an array
      IFS=',' read -ra key_value <<< "${input_cmd[$((i + 1))]}"
      for env_var in "${key_value[@]}"; do
        # Export each environment variable
        IFS='=' read -r key value <<< "$env_var"
        export "$key=$value"
      done
      # Skip the next item since it's part of the --env option
      ((i++))
    elif [[ "${input_cmd[$i]}" == "osp_placeholder.sif" ]]; then
      osp_placeholder_index=$i
      processing_envs=0
      break # Stop processing once osp_placeholder.sif is found
    fi
  done

  # Return the array starting from the element right after osp_placeholder.sif keyword
  if [ $osp_placeholder_index -ne -1 ]; then
    prog_and_args=("${input_cmd[@]:$((osp_placeholder_index + 1))}")
  fi

  echo "${prog_and_args[@]}"
}

find_unused_local_port() {
    local PORTFILE="/tmp/used_ports_shared"
    local START_PORT=3000
    local END_PORT=65000
    local port

    # Ensure the shared PORTFILE exists
    touch "$PORTFILE"

    for port in $(seq $START_PORT $END_PORT); do
        # Check if the port is already reserved or actively in use
        if ! grep -q "^$port\$" "$PORTFILE" && ! ss -tuln | grep -q ":$port "; then
            # Atomically add the port to the PORTFILE
            echo $port >> "$PORTFILE"
            echo $port
            return 0
        fi
    done

    echo "No available ports found in range $START_PORT to $END_PORT" >&2
    exit 1
}

    grep -xFf <(echo "$active_ports") "$PORTFILE" > "$PORTFILE.tmp" && mv "$PORTFILE.tmp" "$PORTFILE"
